import Config from './config';
const { Page } = require('playwright');
const { createLogger, format, transports } = require('winston');
const axios = require('axios');

/**
 * Class representing a login page.
 * @class
 * */

class LoginPage {
    //intègre des rapports et des journaux détaillés à l'aide de la bibliothèque mochawesome pour les rapports HTML
    // et winston pour les journaux
    /**
     * Create a LoginPage.
     * @param {Object} page - The Playwright page object.
     */
    constructor(page) {
        this.page = page;
        this.logger = createLogger({
            level: 'info',
            format: format.combine(
                format.timestamp({ format: 'YYYY-MM-DD HH:mm:ss' }),
                format.printf(({ timestamp, level, message }) => `${timestamp} [${level.toUpperCase()}]: ${message}`)
            ),
            transports: [new transports.Console()]
        });
    }

    /**
     * Navigate to the login page.
     * @async
     */
    async navigateToLoginPage() {
        const url = Config.BASE_URL + Config.LOGIN_ENDPOINT;
        await this.page.goto(url);
        this.logger.info(`Navigated to ${url}`);
    }

    async handlePopin() {
        const popinSelector = '.mainContent';
        const popin = await this.page.waitForSelector(popinSelector, { timeout: 5000 }).catch(() => null);

        if (popin) {
            const refuseButtonSelector = '.pdynamicbutton .required';
            const refuseButton = await this.page.$(refuseButtonSelector);
            await refuseButton.click();
            console.log('Pop-in détecté. Cliqué sur "CONTINUER SANS ACCEPTER"');
        } else {
            console.log('Pop-in non détecté. Continuer le script...');
        }
    }

    /**
     * Fill the login form with credentials.
     * @async
     */
    async fillLoginForm() {
        const emailInputSelector = '#email';
        const passwordInputSelector = 'data-testid="loginPasswordField"';

        await this.page.fill(emailInputSelector, Config.USERNAME);
        await this.page.fill(passwordInputSelector, Config.PASSWORD);
    }

    async submitLoginForm() {
        const submitButtonSelector = 'data-testid="loginSubmitButton"';
        await this.page.click(submitButtonSelector);
    }

    async isLoggedIn() {
        const loggedInUserSelector = 'data-testid="loggedInUser"';
        return await this.page.isVisible(loggedInUserSelector);
    }

    async getLoggedInUser() {
        const loggedInUserSelector = 'data-testid="loggedInUser"';
        return await this.page.textContent(loggedInUserSelector);
    }
}

export default LoginPage;
mocha.reporter('mochawesome');
